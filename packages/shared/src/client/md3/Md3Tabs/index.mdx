## Usuage

```tsx
function Example() {
  const [selected, onSelect] = useState<number | undefined>();
  const options = [
    { label: "Tab1" },
    { label: "Tab2" },
    { label: "Tab3", icon: MdAccessAlarm },
  ];

  return (
    <>
    <Tabs options={options} selectedIndex={selected} onChange={onSelect} />
    {options.map((panel, idx) => {
      return (
        idx === tabIdx && (
          <div key={`homepage-tabpanel-${panel}`} idx={idx}>content</div>
        )
      );
    })}
    </>
  );
}
```
